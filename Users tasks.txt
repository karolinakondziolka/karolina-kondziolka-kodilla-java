SELECT IL.NAME, I.SUMMARY
FROM ISSUESLISTS IL
JOIN ISSUES I ON I.ISSUESLISTS_ID = IL.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY
FROM USERS U
JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_NUMBERS
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY U.LASTNAME, U.FIRSTNAME;
